Ac Code

class NumArray {
public:
    NumArray(vector<int>& nums) {
        for(int i = 0; i < nums.size(); ++i) n[i + 1] = nums[i];
        l = nums.size();
        build(1, l, 1);
    }
    
    void update(int index, int val) {
        updateNode(1, l, 1, index + 1, val);
    }
    
    int sumRange(int left, int right) {
        return sum(1, l, 1, left + 1, right + 1);
    }

private:
    int n[30005];
    int l;
    
    int node[30005 * 3];
    
    void build(int l, int r, int k) {
       if(l == r){
          node[k] = n[l];
          return;
       }
        int mid = (l + r) >> 1;
        build(l, mid, k << 1);
        build(mid + 1, r, k << 1 | 1);
        node[k] = node[k << 1] + node[k << 1 | 1];
    }
    
    void updateNode(int l, int r, int k, int x, int val){
        if(l == r){
            node[k] = val;
            return;
        }
        int mid = (l + r) >> 1;
        if(mid < x) updateNode(mid + 1, r, k << 1 | 1, x, val);
        else updateNode(l, mid, k << 1, x, val);
        node[k] = node[k << 1] + node[k << 1 | 1];
    }
    
    int sum(int l, int r, int k, int L, int R){
        if(L <= l && R >= r) return node[k];
        int ans = 0;
        int mid = (l + r) >> 1;
        if(L <= mid) ans += sum(l, mid, k << 1, L, R);
        if(R > mid) ans += sum(mid + 1, r, k << 1 | 1, L, R);
        return ans;
    }
    
};

/**
 * Your NumArray object will be instantiated and called as such:
 * NumArray* obj = new NumArray(nums);
 * obj->update(index,val);
 * int param_2 = obj->sumRange(left,right);
 */
